Algoritmo AreaTriangulo
    // Ingresar longitudes de los lados
    Leer lado1
    Leer lado2
    Leer lado3

    // Calcular semiperímetro
    s = (lado1 + lado2 + lado3) / 2

    // Calcular área
    area = raiz(s * (s - lado1) * (s - lado2) * (s - lado3))

    // Mostrar resultado
    Escribir "El área del triángulo es: ", area
